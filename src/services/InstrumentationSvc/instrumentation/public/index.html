Rising Tide - REST enabled Instrumentation Service

Available Methods

/logs/:logEntry

Sample usage :

Open up console and paste the following :
GET request :
$.ajax({ url : 'http://localhost:9000/api/logs', type : 'GET', contentType : 'application/json', data : { EventId: "Random", ApplicationName: "RisingTide", FeatureName: "FieldMapping", Category: "Audit", Messages: ["John[PCA] logged in from tool.", "Log in time : 10:30"], TraceLevel: "error" }, success : function(data) { console.log(d); }, error: function(e) { console.log(e.message); } });

POST request :
$.ajax({ url : 'http://localhost:9000/api/logs', type : 'POST', contentType : 'application/json', data : { EventId: "Random", ApplicationName: "RisingTide", FeatureName: "FieldMapping", Category: "Audit", Messages: ["John[PCA] logged in from tool.", "Log in time : 10:30"], TraceLevel: "error" }, success : function(data) { console.log(data); }, error: function(e) { console.log(e.message); } });

JSONP request :
$.ajax({ url : 'http://localhost:9000/api/logs', type : 'GET', contentType : 'application/json', dataType : 'jsonp', data : { EventId: "Random", ApplicationName: "RisingTide", FeatureName: "FieldMapping", Category: "Audit", Messages: ["John[PCA] logged in from tool.", "Log in time : 10:30"], TraceLevel: "error" } , success: function(data) { console.dir(data); }, error: function(e) { console.log(e.message); }, jsonpCallback : 'jsonCallback' });
